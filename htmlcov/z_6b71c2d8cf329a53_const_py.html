131# Recurrence types. 

132REC_FLOATING = "floating" 

133REC_FIXED = "fixed" 

134# A condition-driven task with no schedule. An owner integration arms it (via 

135# ``trigger_task`` or by creating it) when a condition becomes true and clears it 

136# (via ``complete_task``) when the condition resolves. Its ``next_due`` *is* its 

137# state: ``None`` = dormant (invisible to every time surface), a timestamp = 

138# active/due-now. See docs/INTEGRATING.md "Condition-driven (triggered) tasks". 

139REC_TRIGGERED = "triggered" 

140# A user-scheduled do-once task. It carries its own ``due`` datetime (the chosen 

141# due date); ``compute_next_due`` reads it back, and completing the task sets 

142# ``next_due = None`` permanently (no rescheduling) — it goes dormant like a 

143# triggered task, but undoing the completion re-arms it to ``due`` (its state is 

144# history-driven, not condition-driven). See docs/EVENTS.md / README. 

145REC_ONE_OFF = "one-off" 

146# A sensor-based task: Home Keeper derives its armed/dormant state from a bound 

147# numeric sensor rather than the clock. Like ``triggered`` its ``next_due`` *is* its 

148# state (``None`` = dormant, a timestamp = armed/due-now), but Home Keeper itself — 

149# not an external owner — arms it via a pure evaluator fed by the live reading. The 

150# binding lives in ``task["sensor"]`` (see ``models.normalize_sensor`` / 

151# ``sensor_tasks.py``). Two modes: ``usage`` (a meter — due after the reading 

152# advances ``target`` units since the last completion) and ``threshold`` (due when 

153# the reading crosses a comparison). See docs/SENSOR_TASKS_PLAN.md. 

154REC_SENSOR = "sensor" 

155RECURRENCE_TYPES = [REC_FLOATING, REC_FIXED, REC_TRIGGERED, REC_ONE_OFF, REC_SENSOR]

225# ── Companion discovery ────────────────────────────────────────────────────── 

226# A "companion" is another integration that works with Home Keeper. They surface 

227# in the panel's Settings → Companions section so users discover the ecosystem. 

228# Two paths feed the registry (see companions.py / companions_catalog.py): 

229# • Push: a Home-Keeper-aware integration self-registers via the 

230# ``home_keeper.register_companion`` service (Pawsistant, the Battery Notes glue). 

231# • Pull: Home Keeper detects a popular *upstream* (e.g. Battery Notes) from a 

232# curated catalog and *suggests* the glue that bridges it. 

233# 

234# Where the in-memory companion registry lives (survives entry reloads; rebuilt on

240# isn't guaranteed, so companions both register at their own setup *and* listen for 

241# this ping. Carries no data.
