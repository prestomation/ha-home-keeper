384def _pick_localized(table: dict[str, str], language: str | None) -> str: 

385 """Resolve *language* against *table*: exact → case-insensitive → base → English. 

386 

387 Mirrors the panel's ``i18n.resolve`` so a code like ``"pt-BR"`` matches exactly, 

388 ``"en-GB"`` falls back to ``"en"``, and anything unknown lands on English. Pure — 

389 safe to unit-test without a Home Assistant runtime. 

390 """ 

391 lang = language or DEFAULT_LANGUAGE 

392 if lang in table: 

393 return table[lang] 

394 low = lang.lower() 

395 for key, value in table.items(): 

396 if key.lower() == low: 

397 return value 

398 base = low.split("-")[0] 

399 for key, value in table.items(): 

400 if key.lower() == base: 

401 return value 

402 return table[DEFAULT_LANGUAGE]
