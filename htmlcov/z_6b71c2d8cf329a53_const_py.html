13PANEL_STATIC_URL = "/home_keeper_panel" # static path that serves the JS bundle

16PANEL_ICON = "mdi:home-clock" 

17WEBCOMPONENT_NAME = "home-keeper-panel" 

18 

19# Storage. 

20STORAGE_KEY = "home_keeper" 

21STORAGE_VERSION = 1 

22# How many completion timestamps to retain per task. Generous so the panel's task 

23# history shows years of cadence (e.g. 500 monthly completions ≈ 40 years) while 

24# still bounding the stored list. When a task that belongs to an appliance is 

25# deleted, this history is archived onto the appliance (see ``assets.append_task_history``). 

26MAX_COMPLETION_HISTORY = 500 

27 

28# Event fired on the HA event bus whenever a task is completed (from any surface: 

29# the to-do list, a device mark-done button, or the complete_task service). This is 

30# the public, client-agnostic hook other integrations subscribe to in order to mirror 

31# completions. The payload carries the task's opaque ``source`` and ``origin`` verbatim; 

32# Home Keeper never inspects either. See docs/INTEGRATING.md. 

33EVENT_TASK_COMPLETED = f"{DOMAIN}_task_completed" 

34 

35# Event fired when a stock-tracked spare part drops to (or below) its reorder 

36# threshold — either because a wear-part replacement was completed (consuming a 

37# spare) or because stock was manually adjusted down. Lets users automate a 

38# shopping-list add / reorder / notification without Home Keeper owning any 

39# shopping integration. Payload is built in events.low_stock_event_data. 

40EVENT_PART_LOW_STOCK = f"{DOMAIN}_part_low_stock" 

41 

42# Assets / appliances (virtual devices + asset metadata). 

43# A virtual asset device is registered with identifier 

44# (DOMAIN, f"{ASSET_IDENTIFIER_PREFIX}_{asset_id}"); the prefix keeps it from 

45# colliding with the per-task self-owned devices, which key on the bare task id. 

46ASSET_IDENTIFIER_PREFIX = "asset" 

47ASSET_KIND_VIRTUAL = "virtual" # Home Keeper provisions the registry device 

48ASSET_KIND_EXISTING = "existing" # metadata attached to another integration's device 

49ASSET_KINDS = [ASSET_KIND_VIRTUAL, ASSET_KIND_EXISTING] 

50 

51# Structured parts / wear items on an asset. A "consumable" is a stocked spare; a 

52# "wear" item degrades over time and, when given a replacement interval, drives a 

53# maintenance task (reusing the recurrence engine + per-task entities). 

54PART_CONSUMABLE = "consumable" 

55PART_WEAR = "wear" 

56PART_TYPES = [PART_CONSUMABLE, PART_WEAR] 

57 

58# Marker on a task dict identifying it as derived from an asset part, so the 

59# part-task reconciler owns it: ``task["source"] = {"asset_id", "part_id"}``. 

60TASK_SOURCE_PART = "part" 

61 

62 

63# Recurrence types. 

64REC_FLOATING = "floating" 

65REC_FIXED = "fixed" 

66# A condition-driven task with no schedule. An owner integration arms it (via 

67# ``trigger_task`` or by creating it) when a condition becomes true and clears it 

68# (via ``complete_task``) when the condition resolves. Its ``next_due`` *is* its 

69# state: ``None`` = dormant (invisible to every time surface), a timestamp = 

70# active/due-now. See docs/INTEGRATING.md "Condition-driven (triggered) tasks". 

71REC_TRIGGERED = "triggered" 

72RECURRENCE_TYPES = [REC_FLOATING, REC_FIXED, REC_TRIGGERED] 

73

75UNIT_DAYS = "days" 

76UNIT_WEEKS = "weeks" 

77UNIT_MONTHS = "months" 

78UNITS = [UNIT_DAYS, UNIT_WEEKS, UNIT_MONTHS] 

79 

80# Fixed schedule frequencies. 

81FREQ_DAILY = "DAILY" 

82FREQ_WEEKLY = "WEEKLY" 

83FREQ_MONTHLY = "MONTHLY" 

84FREQS = [FREQ_DAILY, FREQ_WEEKLY, FREQ_MONTHLY] 

85 

86# How far ahead the calendar expands fixed occurrences, and a hard iteration cap 

87# to guard against runaway expansion loops.

89 

90# Upper bound on a recurrence interval / wear-part replacement interval. Generous 

91# (e.g. 10000 days ≈ 27 years, 10000 months ≈ 833 years) but low enough to keep 

92# date arithmetic well clear of datetime/timedelta overflow. 

93MAX_INTERVAL = 10_000

99SIGNAL_TASK_CONTRIBUTION = f"{DOMAIN}_task_contribution" # noqa: F841 (reserved)

104TASK_MANAGED_BY = "managed_by"
