1"""Diagnostics support for Home Keeper. 

2 

3Provides a downloadable snapshot (from a device or the config entry) of the tasks 

4and assets Home Keeper manages, to make support/debugging easier. Diagnostics are 

5routinely attached to public GitHub issues, so potentially personal fields are 

6redacted: appliance serial numbers, free-form notes/metadata, and per-completion 

7``who`` (a person entity id) / ``photo`` (an image reference). Schedules and 

8structural data are kept so the dump is still useful for debugging. 

9""" 

10 

11from __future__ import annotations 

12 

13from typing import Any 

14 

15from homeassistant.components.diagnostics import async_redact_data 

16from homeassistant.config_entries import ConfigEntry 

17from homeassistant.core import HomeAssistant 

18from homeassistant.helpers.device_registry import DeviceEntry 

19 

20from . import assets as asset_model 

21from .const import ASSET_IDENTIFIER_PREFIX, DOMAIN 

22from .coordinator import HomeKeeperCoordinator 

23 

24# Redacted by key anywhere in the tasks/assets structure (recursively) before the 

25# snapshot leaves the instance — these can carry personal or identifying data. 

26# Structural/schedule fields (recurrence, due dates, thresholds) are kept so the 

27# dump stays useful for debugging. 

28TO_REDACT = {"serial_number", "notes", "who", "photo"} 

29 

30 

31async def async_get_config_entry_diagnostics( 

32 hass: HomeAssistant, entry: ConfigEntry 

33) -> dict[str, Any]: 

34 """Return diagnostics for the Home Keeper config entry.""" 

35 coordinator: HomeKeeperCoordinator = entry.runtime_data 

36 tasks = coordinator.store.list_tasks() 

37 assets = coordinator.store.list_assets() 

38 return { 

39 "counts": { 

40 "tasks": len(tasks), 

41 "assets": len(assets), 

42 "parts": sum(len(a.get("parts", [])) for a in assets), 

43 "virtual_devices": sum(1 for a in assets if a.get("kind") == "virtual"), 

44 }, 

45 "tasks": async_redact_data(tasks, TO_REDACT), 

46 "assets": async_redact_data(assets, TO_REDACT), 

47 }

50async def async_get_device_diagnostics( 

51 hass: HomeAssistant, entry: ConfigEntry, device: DeviceEntry 

52) -> dict[str, Any]: 

53 """Return the Home Keeper tasks/assets associated with a single device. 

54 

55 Scopes the config-entry dump to the device whose page the download was started 

56 from — its appliance (a virtual asset whose ``asset_<id>`` identifier or 

57 ``device_id`` matches) and every task that lives on it (attached by ``device_id``, 

58 a self-owned task device keyed on the bare task id, or derived from one of the 

59 appliance's parts). 

60 """ 

61 coordinator: HomeKeeperCoordinator = entry.runtime_data 

62 tasks = coordinator.store.list_tasks() 

63 assets = coordinator.store.list_assets() 

64 

65 hk_idents = {ident for domain, ident in device.identifiers if domain == DOMAIN} 

66 device_assets = [ 

67 asset 

68 for asset in assets 

69 if asset.get("device_id") == device.id 

70 or f"{ASSET_IDENTIFIER_PREFIX}_{asset.get('id')}" in hk_idents 

71 ] 

72 device_tasks = [ 

73 task 

74 for task in tasks 

75 if task.get("device_id") == device.id 

76 or task.get("id") in hk_idents 

77 or any( 

78 asset_model.task_relates_to_asset(task, asset) for asset in device_assets 

79 ) 

80 ] 

81 return { 

82 "device": { 

83 "id": device.id, 

84 "name": device.name, 

85 "identifiers": list(hk_idents), 

86 }, 

87 "counts": { 

88 "tasks": len(device_tasks), 

89 "assets": len(device_assets), 

90 "parts": sum(len(a.get("parts", [])) for a in device_assets), 

91 }, 

92 "tasks": async_redact_data(device_tasks, TO_REDACT), 

93 "assets": async_redact_data(device_assets, TO_REDACT), 

94 }
