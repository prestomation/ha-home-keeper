1"""Pure (HA-free) helpers for uploaded asset documents. 

2 

3The on-disk storage and HTTP view live in ``manuals.py`` (which needs Home 

4Assistant); the security-critical, side-effect-free pieces — magic-byte sniffing, 

5filename sanitization, the upload allowlist/size guard, and the path-traversal guard — 

6live here so they stay unit-testable without an HA runtime (see 

7``tests/unit/test_documents.py``). This module imports nothing from Home Assistant. 

8""" 

9 

10from __future__ import annotations 

11 

12import re 

13from pathlib import Path, PurePath 

14 

15from .assets import AssetValidationError 

16from .const import MAX_DOCUMENT_BYTES

18# content-type -> canonical extension. The key set is the upload allowlist. 

19TYPE_EXTENSIONS = { 

20 "application/pdf": ".pdf", 

21 "image/png": ".png", 

22 "image/jpeg": ".jpg", 

23 "image/webp": ".webp", 

24 "image/gif": ".gif", 

25} 

26 

27_MAX_FILENAME_LEN = 120 

28 

29 

30def sniff_content_type(data: bytes) -> str | None: 

31 """Return the allowlisted content type *data* matches by magic bytes, or None.""" 

32 if data.startswith(b"%PDF-"): 

33 return "application/pdf" 

34 if data.startswith(b"\x89PNG\r\n\x1a\n"): 

35 return "image/png" 

36 if data.startswith(b"\xff\xd8\xff"): 

37 return "image/jpeg" 

38 if data[:6] in (b"GIF87a", b"GIF89a"): 

39 return "image/gif" 

40 if data[:4] == b"RIFF" and data[8:12] == b"WEBP": 

41 return "image/webp" 

42 return None

45def safe_filename(name: str, content_type: str) -> str: 

46 """Sanitize an uploaded filename to a plain, extension-correct basename. 

47 

48 Strips any path component, collapses unusual characters, guarantees a non-empty 

49 stem, and forces the canonical extension for *content_type* so the served file is 

50 typed consistently regardless of what the client claimed. 

51 """ 

52 base = PurePath(str(name)).name 

53 base = re.sub(r"[^A-Za-z0-9._-]", "_", base).lstrip(".") 

54 stem = PurePath(base).stem or "document" 

55 stem = stem[:_MAX_FILENAME_LEN] 

56 return f"{stem}{TYPE_EXTENSIONS[content_type]}"

59def validate_upload( 

60 filename: str, declared_type: str | None, data: bytes 

61) -> tuple[str, str]: 

62 """Validate an uploaded blob; return ``(content_type, safe_filename)``. 

63 

64 Raises :class:`AssetValidationError` when the file is empty, over the size ceiling, 

65 or not a recognized allowlisted type (sniffed by magic bytes — the declared MIME is 

66 advisory only). The returned content type is the sniffed one, so the stored metadata 

67 can't be spoofed by a misleading client header. 

68 """ 

69 if not data: 

70 raise AssetValidationError("uploaded file is empty") 

71 if len(data) > MAX_DOCUMENT_BYTES: 

72 raise AssetValidationError( 

73 f"file exceeds the {MAX_DOCUMENT_BYTES // (1024 * 1024)} MB limit" 

74 ) 

75 content_type = sniff_content_type(data) 

76 if content_type is None: 

77 raise AssetValidationError( 

78 "unsupported file type (allowed: PDF, PNG, JPEG, WebP, GIF)" 

79 ) 

80 return content_type, safe_filename(filename, content_type)

83def safe_segment(value: str) -> str: 

84 """Reduce an id/filename to a single safe path segment (defense in depth).""" 

85 seg = re.sub(r"[^A-Za-z0-9._-]", "_", PurePath(str(value)).name).lstrip(".") 

86 if not seg or seg in (".", ".."): 

87 raise AssetValidationError("invalid document path") 

88 return seg

91def resolve_under_root(root: Path, *parts: str) -> Path: 

92 """Join *parts* under *root* and assert the result stays inside it. 

93 

94 Each part is first reduced to a safe single segment, then the fully-resolved path 

95 is checked to be relative to the resolved root — a belt-and-braces guard against a 

96 crafted asset/document id or filename escaping the storage tree (``..``, absolute 

97 paths, separators). 

98 """ 

99 root = root.resolve() 

100 candidate = root.joinpath(*(safe_segment(p) for p in parts)).resolve() 

101 if not candidate.is_relative_to(root): 101 ↛ 102line 101 didn't jump to line 102 because the condition on line 101 was never true

102 raise AssetValidationError("document path escapes the storage root") 

103 return candidate

106def document_path(root: Path, asset_id: str, document_id: str, filename: str) -> Path: 

107 """The on-disk path for an uploaded document, guarded against traversal.""" 

108 return resolve_under_root( 

109 root, asset_id, f"{document_id}__{safe_segment(filename)}"
