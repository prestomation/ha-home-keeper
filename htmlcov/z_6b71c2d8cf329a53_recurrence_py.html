1"""Recurrence engine for Home Keeper. 

2 

3These are deliberately *pure* functions: they take and return timezone-aware 

4``datetime`` objects and plain task dicts, and import nothing from Home Assistant. 

5That keeps the product's core logic trivially unit-testable in isolation (the 

6caller is responsible for passing an aware ``now`` from ``homeassistant.util.dt``). 

7 

8Two recurrence models are supported: 

9 

10* **floating** — the next due date is measured from the last completion: 

11 ``next_due = last_completed + interval·unit``. A task that has *never* been 

12 completed has no clock to measure from, so it is due immediately (``now``) rather 

13 than a full interval into the future — a brand-new chore you haven't done yet is 

14 due now, not "in N days". Completing the task (or seeding an initial completion at 

15 creation) starts the clock. A missed task simply stays overdue; the due date does 

16 not march forward on its own. 

17 

18* **fixed** — the next due date follows a calendar schedule anchored at a fixed 

19 datetime (``FREQ=DAILY|WEEKLY|MONTHLY`` every ``interval`` steps). Completing an 

20 occurrence records history but the schedule advances independently of when the 

21 task was actually completed. 

22""" 

23

44def add_months(dt: datetime, months: int) -> datetime: 

45 """Return *dt* shifted by *months*, clamping the day to the month's length. 

46 

47 e.g. ``add_months(Jan 31, 1) -> Feb 28`` (or Feb 29 in a leap year). The 

48 time-of-day and tzinfo are preserved. 

49 """ 

50 if months == 0: 50 ↛ 51line 50 didn't jump to line 51 because the condition on line 50 was never true

51 return dt 

52 # Convert to a zero-based month index for clean arithmetic. Python's floor 

53 # division and non-negative modulo make this correct for negative months too 

54 # (e.g. Jan + (-2) months -> month_index -2 -> year-1, month 11 = November), 

55 # though in practice this integration only ever advances by positive months. 

56 month_index = dt.month - 1 + months 

57 year = dt.year + month_index // 12 

58 month = month_index % 12 + 1 

59 last_day = _calendar.monthrange(year, month)[1] 

60 day = min(dt.day, last_day) 

61 return dt.replace(year=year, month=month, day=day)

64def add_interval(dt: datetime, interval: int, unit: str) -> datetime: 

65 """Return *dt* advanced by ``interval`` of ``unit`` (days/weeks/months).""" 

66 if interval < 1: 

67 raise ValueError(f"interval must be >= 1, got {interval}") 

68 if unit == UNIT_DAYS: 

69 return dt + timedelta(days=interval) 

70 if unit == UNIT_WEEKS: 

71 return dt + timedelta(weeks=interval) 

72 if unit == UNIT_MONTHS: 

73 return add_months(dt, interval) 

74 raise ValueError(f"unknown unit: {unit!r}")

107def _fast_forward(anchor: datetime, freq: str, interval: int, target: datetime) -> datetime: 

108 """An occurrence at or just before *target*, jumped to in O(1). 

109 

110 A long-dormant fixed schedule can be thousands of steps past its anchor; 

111 stepping one occurrence at a time would be slow (and historically raised once 

112 it blew an iteration cap). We deliberately *under*-shoot (``- 1`` step) so the 

113 caller's short loop finishes on the exact occurrence using wall-clock stepping 

114 — correct across DST and month-length clamping. Day/week deltas are exact, so 

115 the bulk jump is too; MONTHLY is left to the loop (it would need ~830 years to 

116 approach the cap) to preserve its progressive day-clamping behavior. 

117 """ 

118 elapsed = (target - anchor).total_seconds() 

119 if elapsed <= 0: 119 ↛ 120line 119 didn't jump to line 120 because the condition on line 119 was never true

120 return anchor 

121 if freq == FREQ_DAILY: 

122 steps = max(0, int(elapsed // (86_400 * interval)) - 1) 

123 return anchor + timedelta(days=interval * steps) 

124 if freq == FREQ_WEEKLY: 

125 steps = max(0, int(elapsed // (604_800 * interval)) - 1) 

126 return anchor + timedelta(weeks=interval * steps) 

127 return anchor

199def compute_next_due(task: dict, *, now: datetime) -> datetime: 

200 """Compute next_due for *task* from its current state (no mutation).""" 

201 rec_type = task.get("recurrence_type", REC_FLOATING) 

202 if rec_type == REC_FLOATING: 

203 return compute_floating_next_due( 

204 _parse(task.get("last_completed")), 

205 int(task["interval"]), 

206 task["unit"], 

207 now=now, 

208 ) 

209 if rec_type == REC_FIXED: 

210 anchor = _parse(task["anchor"]) 

211 assert anchor is not None 

212 return next_fixed_occurrence( 

213 anchor, task["freq"], int(task["interval"]), after=now 

214 )

224def apply_completion(task: dict, completed_at: datetime, *, now: datetime) -> dict: 

225 """Return *task* mutated to reflect a completion at *completed_at*. 

226 

227 Records the completion in history (capped) and recomputes ``next_due``: 

228 

229 * floating -> measured from ``completed_at`` (the clock resets) 

230 * fixed -> the next scheduled occurrence after ``now`` (schedule-driven; 

231 completion only marks the occurrence done) 

232 * triggered -> ``next_due = None`` (dormant): completing a condition-driven 

233 task *clears* the condition rather than rescheduling it, so it leaves every 

234 time surface until the owner re-arms it. History is still recorded, so the 

235 replacement cadence accumulates on the task. 

236 """ 

237 history = list(task.get("completions", [])) 

238 history.append({"ts": completed_at.isoformat()}) 

239 if len(history) > MAX_COMPLETION_HISTORY: 239 ↛ 240line 239 didn't jump to line 240 because the condition on line 239 was never true

240 history = history[-MAX_COMPLETION_HISTORY:] 

241 task["completions"] = history 

242 task["last_completed"] = completed_at.isoformat() 

243 

244 rec_type = task.get("recurrence_type", REC_FLOATING) 

245 if rec_type == REC_FLOATING: 

246 task["next_due"] = compute_floating_next_due( 

247 completed_at, int(task["interval"]), task["unit"], now=now 

248 ).isoformat() 

249 elif rec_type == REC_FIXED: 

250 anchor = _parse(task["anchor"]) 

251 assert anchor is not None 

252 task["next_due"] = next_fixed_occurrence( 

253 anchor, task["freq"], int(task["interval"]), after=now 

254 ).isoformat() 

255 elif rec_type == REC_TRIGGERED: 255 ↛ 258line 255 didn't jump to line 258 because the condition on line 255 was always true

256 task["next_due"] = None 

257 else: 

258 raise ValueError(f"unknown recurrence_type: {rec_type!r}") 

259 return task

262def remove_completion(task: dict, ts: str, *, now: datetime) -> dict: 

263 """Return *task* with the completion at ISO timestamp *ts* removed. 

264 

265 Undoes an accidental completion: drops the first matching history entry, 

266 re-derives ``last_completed`` from the remaining history (the latest, or None), 

267 and recomputes ``next_due`` from that state. For a floating task this rewinds 

268 the clock to the prior completion; for a fixed task ``next_due`` stays 

269 schedule-driven; for a triggered task ``next_due`` is left untouched (its 

270 armed/dormant state is condition-driven, not history-driven — editing the 

271 replacement log must not arm a dormant task). A no-op when *ts* is not present. 

272 """ 

273 history = list(task.get("completions", [])) 

274 for index, entry in enumerate(history): 

275 if entry.get("ts") == ts: 

276 del history[index] 

277 break 

278 task["completions"] = history 

279 if history: 

280 latest = max(history, key=lambda entry: _parse(entry["ts"])) 

281 task["last_completed"] = latest["ts"] 

282 else: 

283 task["last_completed"] = None 

284 if task.get("recurrence_type") != REC_TRIGGERED: 

285 task["next_due"] = compute_next_due(task, now=now).isoformat() 

286 return task

289def is_overdue(task: dict, *, now: datetime) -> bool: 

290 """True when the task's next due date is at or before *now*. 

291 

292 Convention: the due instant itself counts as overdue (``>=``). This pairs with 

293 :func:`is_due_soon`'s strict ``now < next_due`` lower bound so the two states 

294 partition cleanly at the boundary (no gap, no overlap). This is intentionally 

295 *not* the strict ``>`` used by :func:`next_fixed_occurrence`, which serves a 

296 different purpose (advancing to the next occurrence); a completion always sets 

297 ``next_due`` strictly in the future, so a just-completed task is never overdue. 

298 """ 

299 next_due = _parse(task.get("next_due")) 

300 if next_due is None: 

301 return False 

302 return now >= next_due
