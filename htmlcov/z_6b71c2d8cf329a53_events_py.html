14def task_event_data( 

15 task: dict[str, Any], *, extra: dict[str, Any] | None = None 

16) -> dict[str, Any]: 

17 """Return the common "spine" payload shared by every task event. 

18 

19 One template works across `home_keeper_task_created` / `_updated` / `_deleted` / 

20 `_completed` / `_uncompleted` / `_triggered` / `_overdue` / `_due_soon`: the same 

21 identity (``task_id``/``name``), attachment (``device_id``/``area_id``/``labels``), 

22 schedule state (``recurrence_type``/``next_due``/``enabled``), and the opaque 

23 ``source`` / well-known ``managed_by`` blocks Home Keeper echoes verbatim. 

24 ``device_id`` is the task's stored registry device id (already a registry id; 

25 echoed as-is, or ``None`` when the task isn't attached to a device — its per-task 

26 entities then live on a self-owned device keyed on ``task_id``). Per-event extras 

27 (``changed_fields``, 

28 ``days_overdue``, ``due_in_hours``) are merged in via *extra*. 

29 """ 

30 data: dict[str, Any] = { 

31 "task_id": task.get("id"), 

32 "name": task.get("name"), 

33 "device_id": task.get("device_id"), 

34 "area_id": task.get("area_id"), 

35 "recurrence_type": task.get("recurrence_type"), 

36 "next_due": task.get("next_due"), 

37 "enabled": task.get("enabled", True), 

38 "labels": task.get("labels", []), 

39 "source": task.get("source"), 

40 "managed_by": task.get("managed_by"), 

41 "task_chips": task.get("task_chips", []), 

42 } 

43 if extra: 43 ↛ 45line 43 didn't jump to line 45 because the condition on line 43 was always true

44 data.update(extra) 

45 return data

48def completion_event_data( 

49 task: dict[str, Any], 

50 when: Any, 

51 origin: str | None, 

52 *, 

53 metadata: dict[str, Any] | None = None, 

54) -> dict[str, Any]: 

55 """Return the ``home_keeper_task_completed`` event data for a completed task. 

56 

57 The common task spine plus the completion-specific ``completed_at`` and ``origin``. 

58 ``when`` is the completion datetime (or an ISO string); ``origin`` is the opaque 

59 caller marker. The task passed in is the *post-completion* task, so ``next_due`` is 

60 already the next occurrence (``None`` for a now-dormant triggered task). 

61 

62 *metadata* is the cleaned per-completion context that was recorded 

63 (``note``/``cost``/``photo``/``who``); only the keys that carried a value are 

64 merged in, so a plain one-click completion adds nothing beyond the spine. 

65 """ 

66 extra: dict[str, Any] = { 

67 "completed_at": when.isoformat() if hasattr(when, "isoformat") else when, 

68 "origin": origin, 

69 } 

70 if metadata: 

71 extra.update(metadata) 

72 return task_event_data(task, extra=extra)

75def asset_event_data( 

76 asset: dict[str, Any], *, extra: dict[str, Any] | None = None 

77) -> dict[str, Any]: 

78 """Return the common payload for `home_keeper_asset_*` lifecycle events. 

79 

80 Carries the appliance identity and its registry ``device_id`` (``None`` until a 

81 virtual asset's device is provisioned). Per-event extras (``changed_fields`` for an 

82 update) merge in via *extra*. 

83 """ 

84 data: dict[str, Any] = { 

85 "asset_id": asset.get("id"), 

86 "asset_name": asset.get("name") or "", 

87 "device_id": asset.get("device_id"), 

88 } 

89 if extra: 

90 data.update(extra) 

91 return data

94def stock_event_data(asset: dict[str, Any], part: dict[str, Any]) -> dict[str, Any]: 

95 """Return the shared payload for the three `home_keeper_part_*` stock events. 

96 

97 Low-stock, out-of-stock and restocked all carry the same shape so an automation 

98 template is interchangeable across them. Carries enough to drive an automation 

99 (notify, add to a shopping list, reorder) without re-querying: which appliance and 

100 part, the part/vendor identifiers needed to rebuy, and the current vs. threshold 

101 counts. 

102 """ 

103 return { 

104 "asset_id": asset.get("id"), 

105 "asset_name": asset.get("name") or "", 

106 "device_id": asset.get("device_id"), 

107 "part_id": part.get("id"), 

108 "part_name": part.get("name") or "", 

109 "part_number": part.get("part_number") or "", 

110 "vendor": part.get("vendor") or "", 

111 "stock": part.get("stock"), 

112 "reorder_at": part.get("reorder_at"), 

113 }
