1"""Asset (appliance) model helpers for Home Keeper. 

2 

3An *asset* is a physical thing you keep — a fridge, a furnace, an under-sink RO 

4unit — that maintenance tasks (and, later, batteries / other metadata) hang off 

5of. Like tasks, an asset is stored as a plain JSON-serializable ``dict`` so it 

6round-trips through HA's ``Store`` and the websocket/service APIs untouched. 

7 

8Concerns living in one record (see ``IDEAS.md`` / ``docs/DESIGN.md``): 

9 

101. **Virtual-device provision** — when an appliance has no Home Assistant device 

11 to attach to, Home Keeper registers a real registry device for it 

12 (``kind == "virtual"``). Its registry identifier is 

13 ``(DOMAIN, f"{ASSET_IDENTIFIER_PREFIX}_{asset_id}")`` so it never collides with 

14 the per-task self-owned devices, which key on the bare task id. 

152. **Asset metadata** — descriptive/temporal attributes (purchase/install/warranty 

16 dates, cost, vendor, manual link, notes) that can also enrich an *existing* 

17 device from another integration (``kind == "existing"``), without owning it. 

183. **Parts / wear items** — a structured list of components (filters, bulbs, anode 

19 rods, shade material). A ``wear`` part with a replacement interval drives a 

20 maintenance task elsewhere (see ``store.reconcile_part_tasks``). 

214. **Relationships** — ``parent_asset_id`` makes a virtual asset a *subdevice* of 

22 another virtual asset (native ``via_device`` hierarchy); ``related_device_ids`` 

23 loosely associates arbitrary registry devices (including foreign ones HA won't 

24 let us reparent), surfaced only in the panel. 

25 

26This module imports nothing from Home Assistant so it stays unit-testable; HA 

27specifics (the current time, the area/device registries) are injected or validated 

28by the caller. 

29"""

232def normalize_fields(data: dict) -> dict: 

233 """Validate and normalize the user-supplied fields of an asset. 

234 

235 Returns only the descriptive/provisioning fields (no id / created / device_id / 

236 identifiers — those are assigned by :func:`build_asset` and the provisioning 

237 layer). ``kind`` defaults to ``virtual``. 

238 """ 

239 kind = data.get("kind", ASSET_KIND_VIRTUAL) 

240 if kind not in ASSET_KINDS: 

241 raise AssetValidationError(f"invalid kind: {kind!r}") 

242 

243 fields: dict[str, Any] = {"kind": kind, "area_id": data.get("area_id") or None} 

244 

245 if kind == ASSET_KIND_VIRTUAL: 

246 # We own this device, so a name is required (it titles the device page).

252 fields["parent_asset_id"] = data.get("parent_asset_id") or None

254 # The device supplies its own name; we only need to know which device.

279def build_asset(data: dict, *, now: datetime) -> dict: 

280 """Create a brand-new asset dict (with id, created, and provisioning anchors).""" 

281 fields = normalize_fields(data) 

282 asset_id = str(uuid.uuid4()) 

283 asset: dict[str, Any] = { 

284 "id": asset_id, 

285 "created": now.isoformat(), 

286 # device_id is the registry anchor: for an existing-device asset it is the 

287 # user-chosen target; for a virtual asset it is filled in once the device 

288 # is provisioned. identifiers/connections snapshot the device for 

289 # reconciliation when another integration recreates it under a new id. 

290 "device_id": fields.pop("device_id", None), 

291 "identifiers": [], 

292 "connections": [], 

293 **fields, 

294 } 

295 if asset["kind"] == ASSET_KIND_VIRTUAL: 

296 asset["identifiers"] = [list(asset_device_identifier(asset_id))] 

297 return asset

370# ── task ↔ appliance association & completion-history retention ─────────────── 

371# These pure helpers answer "which tasks belong to this appliance" and preserve a 

372# deleted task's completion history on its appliance. The retention rule is 

373# reference-counting: a task's history outlives the task only while an appliance 

374# still references it (a part-derived, device-attached, or related-device task). A 

375# standalone task's history is dropped with it, and deleting an appliance drops the 

376# archive with it (it is a field on the asset record).

468 """Remove a single archived completion (ISO *ts*) from *asset*'s history. 

469

475 for index, entry in enumerate(history):

484 else: 

485 del history[index]
