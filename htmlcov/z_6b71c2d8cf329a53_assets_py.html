1"""Asset (appliance) model helpers for Home Keeper. 

2 

3An *asset* is a physical thing you keep — a fridge, a furnace, an under-sink RO 

4unit — that maintenance tasks (and, later, batteries / other metadata) hang off 

5of. Like tasks, an asset is stored as a plain JSON-serializable ``dict`` so it 

6round-trips through HA's ``Store`` and the websocket/service APIs untouched. 

7 

8Concerns living in one record (see ``IDEAS.md`` / ``docs/DESIGN.md``): 

9 

101. **Virtual-device provision** — when an appliance has no Home Assistant device 

11 to attach to, Home Keeper registers a real registry device for it 

12 (``kind == "virtual"``). Its registry identifier is 

13 ``(DOMAIN, f"{ASSET_IDENTIFIER_PREFIX}_{asset_id}")`` so it never collides with 

14 the per-task self-owned devices, which key on the bare task id. 

152. **Asset metadata** — descriptive/temporal attributes (purchase/install/warranty 

16 dates, cost, vendor, manual link, notes) that can also enrich an *existing* 

17 device from another integration (``kind == "existing"``), without owning it. 

183. **Parts / wear items** — a structured list of components (filters, bulbs, anode 

19 rods, shade material). A ``wear`` part with a replacement interval drives a 

20 maintenance task elsewhere (see ``store.reconcile_part_tasks``). 

214. **Relationships** — ``parent_asset_id`` makes a virtual asset a *subdevice* of 

22 another virtual asset (native ``via_device`` hierarchy); ``related_device_ids`` 

23 loosely associates arbitrary registry devices (including foreign ones HA won't 

24 let us reparent), surfaced only in the panel. 

25 

26This module imports nothing from Home Assistant so it stays unit-testable; HA 

27specifics (the current time, the area/device registries) are injected or validated 

28by the caller. 

29"""

201def _merge_parts(existing: list[dict], incoming: list[dict]) -> list[dict]: 

202 """Carry backend-managed part fields (``last_replaced``) across an edit. 

203 

204 The panel submits the full parts list; for parts that already exist (matched by 

205 ``id``) we keep the stored ``last_replaced`` unless the caller explicitly set it. 

206 """ 

207 by_id = {p["id"]: p for p in existing} 

208 merged: list[dict] = [] 

209 for part in incoming: 

210 prior = by_id.get(part["id"]) 

211 if prior and part.get("last_replaced") is None: 211 ↛ 213line 211 didn't jump to line 213 because the condition on line 211 was always true

213 merged.append(part)

217def normalize_fields(data: dict) -> dict: 

218 """Validate and normalize the user-supplied fields of an asset. 

219 

220 Returns only the descriptive/provisioning fields (no id / created / device_id / 

221 identifiers — those are assigned by :func:`build_asset` and the provisioning 

222 layer). ``kind`` defaults to ``virtual``. 

223 """ 

224 kind = data.get("kind", ASSET_KIND_VIRTUAL) 

225 if kind not in ASSET_KINDS: 

226 raise AssetValidationError(f"invalid kind: {kind!r}") 

227 

228 fields: dict[str, Any] = {"kind": kind, "area_id": data.get("area_id") or None} 

229 

230 if kind == ASSET_KIND_VIRTUAL: 

231 # We own this device, so a name is required (it titles the device page). 

232 name = str(_require(data, "name")).strip() 

233 if not name: 233 ↛ 234line 233 didn't jump to line 234 because the condition on line 233 was never true

234 raise AssetValidationError("name must not be empty") 

235 fields["name"] = name 

236 # Only a device we own can be a native subdevice of another via via_device. 

237 fields["parent_asset_id"] = data.get("parent_asset_id") or None 

238 else: # ASSET_KIND_EXISTING — metadata attached to someone else's device. 

239 # The device supplies its own name; we only need to know which device. 

240 fields["name"] = str(data.get("name", "")).strip() 

241 fields["device_id"] = _require(data, "device_id") 

242 fields["parent_asset_id"] = None 

243 

244 for key in _TEXT_FIELDS: 

245 value = data.get(key) 

246 fields[key] = str(value).strip() if value not in (None, "") else "" 

247 

248 fields["manual_url"] = _normalize_url(data.get("manual_url")) 

249 fields["icon"] = _normalize_icon(data.get("icon")) 

250 

251 for key in DATE_FIELDS: 

252 fields[key] = _normalize_date(data.get(key), key) 

253 

254 fields["cost"] = _normalize_cost(data.get("cost")) 

255 fields["parts"] = _normalize_parts(data.get("parts")) 

256 

257 related = data.get("related_device_ids") or []

260 fields["related_device_ids"] = [str(d) for d in related if d] 

261 return fields

264def build_asset(data: dict, *, now: datetime) -> dict: 

265 """Create a brand-new asset dict (with id, created, and provisioning anchors).""" 

266 fields = normalize_fields(data) 

267 asset_id = str(uuid.uuid4()) 

268 asset: dict[str, Any] = { 

269 "id": asset_id, 

270 "created": now.isoformat(), 

271 # device_id is the registry anchor: for an existing-device asset it is the 

272 # user-chosen target; for a virtual asset it is filled in once the device 

273 # is provisioned. identifiers/connections snapshot the device for 

274 # reconciliation when another integration recreates it under a new id. 

275 "device_id": fields.pop("device_id", None), 

276 "identifiers": [], 

277 "connections": [], 

278 **fields, 

279 } 

280 if asset["kind"] == ASSET_KIND_VIRTUAL: 

281 asset["identifiers"] = [list(asset_device_identifier(asset_id))] 

282 return asset

285def merge_update(existing: dict, updates: dict, *, now: datetime) -> dict: 

286 """Return *existing* updated with *updates*, re-validating the whole record. 

287 

288 ``kind`` and the virtual-device identifier are immutable after creation — an 

289 asset cannot switch between owning a device and decorating someone else's. 

290 """ 

291 candidate: dict[str, Any] = { 

292 "kind": existing.get("kind", ASSET_KIND_VIRTUAL), 

293 "name": updates.get("name", existing.get("name")), 

294 "area_id": updates.get("area_id", existing.get("area_id")), 

295 "device_id": updates.get("device_id", existing.get("device_id")), 

296 "cost": updates.get("cost", existing.get("cost")), 

297 "icon": updates.get("icon", existing.get("icon")), 

298 "parent_asset_id": updates.get( 

299 "parent_asset_id", existing.get("parent_asset_id") 

300 ), 

301 "parts": updates.get("parts", existing.get("parts", [])), 

302 "related_device_ids": updates.get( 

303 "related_device_ids", existing.get("related_device_ids", []) 

304 ), 

305 } 

306 for key in (*_TEXT_FIELDS, "manual_url", *DATE_FIELDS): 

307 candidate[key] = updates.get(key, existing.get(key)) 

308 

309 fields = normalize_fields(candidate) 

310 # Preserve backend-managed part fields across the edit. 

311 if "parts" in updates: 

312 fields["parts"] = _merge_parts(existing.get("parts", []), fields["parts"]) 

313 

314 merged = dict(existing) 

315 # For a virtual asset, normalize_fields omits device_id, so the provisioned 

316 # device_id is preserved. For an existing-device asset it carries through the 

317 # (re-)chosen target. Either way the kind and virtual identifier are immutable. 

318 merged.update(fields) 

319 merged["kind"] = existing.get("kind", ASSET_KIND_VIRTUAL) 

320 return merged
