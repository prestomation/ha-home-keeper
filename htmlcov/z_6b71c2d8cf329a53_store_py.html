78def _reject_synced_problem(task: dict[str, Any], origin: str | None) -> None: 

79 """Raise unless *origin* authorizes mutating a problem-sensor-synced task. 

80 

81 A synced task mirrors a ``device_class: problem`` binary sensor and is owned by 

82 the sync. Every user-facing surface (to-do, button, service, websocket, panel) 

83 calls without :data:`ORIGIN_PROBLEM_SENSOR_SYNC`, so this rejects them with a 

84 clear message; only the internal sync (which passes the marker) may arm or clear 

85 the task. The problem itself has to be resolved in the originating integration. 

86 """ 

87 if _problem_source(task) is None or origin == ORIGIN_PROBLEM_SENSOR_SYNC: 

88 return 

89 entity_id = _problem_entity(task) or "the originating integration" 

90 raise models.TaskValidationError( 

91 f"This task mirrors the problem sensor {entity_id} and can't be cleared in " 

92 "Home Keeper. Resolve the problem in the originating integration — the task " 

93 "clears automatically when the sensor returns to OK." 

94 )

97def _changed_fields(before: dict[str, Any], after: dict[str, Any]) -> list[str]: 

98 """Top-level keys whose value differs between *before* and *after*. 

99 

100 Drives the ``changed_fields`` payload on the update events (and the decision to 

101 suppress a no-op update event entirely). Backend-managed bookkeeping keys that 

102 aren't user-meaningful are ignored so a reschedule doesn't spam ``next_due``/ 

103 ``completions`` churn as "changes". 

104 """ 

105 ignore = {"completions", "last_completed", "next_due", "created"} 

106 keys = (set(before) | set(after)) - ignore 

107 return sorted(k for k in keys if before.get(k) != after.get(k))

119 async def load(self) -> None: 

120 """Load tasks and assets from disk (no-op safe on first run). 

121 

122 The ``assets`` key is additive — documents written before assets existed

125 data = await self._store.async_load()

128 else: 

129 self._tasks = {} 

130 if data and isinstance(data.get("assets"), dict): 

131 self._assets = data["assets"] 

132 else: 

133 self._assets = {} 

134 # Additive migrations (no storage-version bump): fold a legacy 

135 # ``part_numbers`` string into structured ``parts`` and drop links to 

136 # assets that no longer exist. 

137 changed = False 

138 for asset in self._assets.values(): 

139 if assets.migrate_legacy_part_numbers(asset): 

140 changed = True 

141 if assets.migrate_documents_from_manual_url(asset): 

142 changed = True 

143 if self._clean_relationship_links(): 

144 changed = True 

145 if changed: 

146 await self._save()

178 async def add_task(self, data: dict[str, Any]) -> dict[str, Any]: 

179 """Create and persist a new task; returns the created task dict. 

180 

181 ``source`` is opaque provenance a caller may attach, but the ``part``, 

182 ``problem_sensor`` and ``buy`` namespaces are *reserved* — the reconcilers own 

183 tasks that carry them (and will silently delete a task whose reserved source 

184 doesn't match a live part/sensor). Reject them here so an external ``add_task`` 

185 can't mint a task that masquerades as reconciler-owned and then vanishes on the 

186 next reconcile pass. Home Keeper's own reconcilers build such tasks via 

187 ``models.build_task`` directly, not through this service path. 

188 """ 

189 source = data.get("source") 

190 if isinstance(source, dict): 

191 reserved = { 

192 TASK_SOURCE_PART, 

193 TASK_SOURCE_PROBLEM_SENSOR, 

194 TASK_SOURCE_BUY, 

195 } & set(source) 

196 if reserved: 

197 raise models.TaskValidationError( 

198 f"source keys {sorted(reserved)} are reserved for Home Keeper's " 

199 "own task reconcilers and cannot be set via add_task" 

200 ) 

201 task = models.build_task(data, now=dt_util.now()) 

202 self._tasks[task["id"]] = task 

203 await self._save() 

204 _LOGGER.debug("Added task %s (%s)", task["id"], task["name"]) 

205 self._hass.bus.async_fire(EVENT_TASK_CREATED, events.task_event_data(task)) 

206 return task

222 EVENT_TASK_UPDATED, 

223 events.task_event_data(merged, extra={"changed_fields": changed}), 

224 ) 

225 return merged

227 async def set_task_consumable( 

228 self, task_id: str, asset_id: str | None, part_id: str | None 

229 ) -> dict[str, Any]: 

230 """Link a task to an asset consumable/part, or clear the link. 

231 

232 A manual link sets the task's ``source`` to a part reference (flagged 

233 ``manual`` so the part-task reconciler leaves it alone — see 

234 ``reconcile.is_manual_part_link``) so that *completing* the task consumes one 

235 spare from the part's ``stock``, firing the edge-triggered low/out-of-stock 

236 events when the reorder threshold is crossed — exactly like a wear-part 

237 replacement. This is how a user wires an arbitrary task (e.g. a sensor task 

238 armed by a fridge's filter-life entity) to the consumable it depletes. 

239 

240 Pass both ``asset_id`` and ``part_id`` to link; pass both as ``None`` to clear. 

241 Raises ``KeyError`` for an unknown task; ``TaskValidationError`` for an unknown 

242 asset/part, a half-specified link, or a task already owned by another source (a 

243 reconciler-derived wear-part task, or a synced ``problem`` sensor). 

244 """ 

245 existing = self._tasks.get(task_id) 

246 if existing is None: 

247 raise KeyError(task_id) 

248 _reject_synced_problem(existing, None) 

249 # A reconciler-derived part task is already bound to its part; re-pointing it by 

250 # hand would just be undone on the next reconcile. Only a user-owned task (no 

251 # part source) or an existing manual link may be (re)linked or cleared. 

252 src = _part_source(existing) 

253 if src is not None and not src.get("manual"): 

254 raise models.TaskValidationError( 

255 "This task is auto-generated from an appliance wear part and is " 

256 "already linked to it — manage its part in the appliance editor." 

257 ) 

258 

259 if asset_id is None and part_id is None: 

260 if existing.get("source") is None: 

261 return existing # already unlinked — no-op, no event 

262 existing["source"] = None 

263 else: 

264 if not asset_id or not part_id: 

265 raise models.TaskValidationError( 

266 "linking a consumable needs both asset_id and part_id" 

267 ) 

268 asset = self._assets.get(asset_id) 

269 if asset is None: 

270 raise models.TaskValidationError(f"unknown asset: {asset_id!r}") 

271 if not any(p.get("id") == part_id for p in asset.get("parts", [])): 

272 raise models.TaskValidationError( 

273 f"asset {asset_id!r} has no part {part_id!r}" 

274 ) 

275 new_source = { 

276 TASK_SOURCE_PART: { 

277 "asset_id": asset_id, 

278 "part_id": part_id, 

279 "manual": True, 

280 } 

281 } 

282 if existing.get("source") == new_source: 

283 return existing # already linked to this part — no-op, no event 

284 existing["source"] = new_source 

285 await self._save() 

286 _LOGGER.debug( 

287 "Set consumable link for task %s -> %s", task_id, existing.get("source") 

288 ) 

289 self._hass.bus.async_fire( 

290 EVENT_TASK_UPDATED, 

291 events.task_event_data(existing, extra={"changed_fields": ["source"]}), 

292 ) 

293 return existing

295 async def trigger_task( 

296 self, task_id: str, *, origin: str | None = None 

297 ) -> dict[str, Any]: 

298 """Arm a condition-driven (``triggered``) or sensor-based task so it's due-now. 

299 

300 Sets ``next_due`` to now, moving the task from dormant to active. This is the 

301 owner-facing counterpart to ``complete_task`` (which clears it back to 

302 dormant): an integration calls it when the condition it monitors becomes true 

303 again (e.g. a battery drops low after a prior replacement), and the sensor-task 

304 watcher calls it when a bound reading meets the task's condition. Idempotent — 

305 arming an already-active task just refreshes its trigger time. Only valid for 

306 ``triggered`` / ``sensor`` tasks; rejects others so callers don't accidentally 

307 strand a scheduled task on a fixed due date. 

308 """ 

309 existing = self._tasks.get(task_id) 

310 if existing is None: 

311 raise KeyError(task_id) 

312 if existing.get("recurrence_type") not in (REC_TRIGGERED, REC_SENSOR): 

313 raise models.TaskValidationError( 

314 "trigger_task is only valid for triggered or sensor-based tasks" 

315 ) 

316 _reject_synced_problem(existing, origin) 

317 existing["next_due"] = dt_util.now().isoformat() 

318 await self._save() 

319 _LOGGER.debug("Triggered task %s (armed, due now)", task_id) 

320 self._hass.bus.async_fire( 

321 EVENT_TASK_TRIGGERED, events.task_event_data(existing) 

322 ) 

323 return existing

325 async def snooze_task( 

326 self, task_id: str, until: Any, *, origin: str | None = None 

327 ) -> dict[str, Any]: 

328 """Push a task's ``next_due`` to *until* without recording a completion. 

329 

330 "Remind me later": unlike :meth:`complete_task` this advances *only* 

331 ``next_due`` — recurrence, ``last_completed`` and the completion history are 

332 untouched — so the schedule isn't advanced, just deferred. Because 

333 ``next_due`` changes, the coordinator re-arms the edge-triggered 

334 overdue/due-soon events for the new date (a fresh reminder fires when the 

335 snooze lapses). *until* is a timezone-aware datetime (the caller computes it 

336 from the requested duration). Rejects a synced problem-sensor task like every 

337 other user mutation, and a **dormant** task (``next_due is None``) — there's no 

338 due date to defer. Fires ``home_keeper_task_snoozed``. 

339 """ 

340 existing = self._tasks.get(task_id) 

341 if existing is None: 

342 raise KeyError(task_id) 

343 _reject_synced_problem(existing, origin) 

344 if existing.get("next_due") is None: 

345 # A dormant task (a completed one-off, or a condition/sensor task not yet 

346 # armed) has no due date to defer; snoozing it would silently re-arm 

347 # something that was intentionally off every time surface. 

348 raise models.TaskValidationError( 

349 "This task is dormant (no due date) — snooze only defers a task that " 

350 "is currently scheduled. Re-arm it instead (undo a completion, or wait " 

351 "for its condition/sensor)." 

352 ) 

353 existing["next_due"] = until.isoformat() 

354 await self._save() 

355 _LOGGER.debug("Snoozed task %s until %s", task_id, existing["next_due"]) 

356 self._hass.bus.async_fire( 

357 EVENT_TASK_SNOOZED, 

358 events.task_event_data( 

359 existing, extra={"snoozed_until": existing["next_due"]} 

360 ), 

361 ) 

362 return existing

364 async def skip_task( 

365 self, task_id: str, *, origin: str | None = None 

366 ) -> dict[str, Any]: 

367 """Advance a task to its next occurrence with **no** completion recorded. 

368 

369 "Skip this one": delegates the (pure) recurrence math to 

370 :func:`recurrence.skip_occurrence` — floating jumps a fresh interval, fixed 

371 advances one scheduled occurrence, and one-off/triggered/sensor go dormant — 

372 without stamping history or ``last_completed``. Rejects a synced 

373 problem-sensor task. Fires ``home_keeper_task_skipped``. 

374 """ 

375 existing = self._tasks.get(task_id) 

376 if existing is None: 

377 raise KeyError(task_id) 

378 _reject_synced_problem(existing, origin) 

379 updated = recurrence.skip_occurrence(dict(existing), now=dt_util.now()) 

380 self._tasks[task_id] = updated 

381 await self._save() 

382 _LOGGER.debug("Skipped task %s; next due %s", task_id, updated.get("next_due")) 

383 self._hass.bus.async_fire(EVENT_TASK_SKIPPED, events.task_event_data(updated)) 

384 return updated

408 async def delete_task(self, task_id: str, *, force: bool = False) -> None: 

409 task = self._tasks.get(task_id) 

410 if task is not None and _part_source(task) and not _is_manual_part_link(task): 

411 # Derived from a wear part; deleting it here would just be recreated by 

412 # the next reconcile. Direct the user to manage the part instead. A *manual* 

413 # consumable link is user-owned, so it is freely deletable (the link is just 

414 # dropped along with the task). 

415 raise models.TaskValidationError( 

416 "This task is managed by an appliance wear part; remove or change " 

417 "the part to delete it." 

418 ) 

419 if task is not None and _buy_source(task): 

420 # System-managed auto-buy reminder; the reconciler would recreate it while 

421 # the part is still low. Direct the user to restock or turn off the option. 

422 raise models.TaskValidationError( 

423 "This is an auto-created buy reminder; restock the part or turn off " 

424 "its auto-buy option to remove it." 

425 ) 

426 if task is not None: 

427 managed_by = task.get("managed_by") 

428 orphaned = self.managed_task_orphaned(task) 

429 if models.deletion_blocked(task, orphaned=orphaned, force=force): 

430 display_name = (managed_by or {}).get( 

431 "display_name" 

432 ) or "an integration" 

433 raise models.TaskValidationError( 

434 f"This task is managed by {display_name}. " 

435 f"Delete it from {display_name} instead." 

436 ) 

437 if task_id in self._tasks: 

438 removed = self._tasks[task_id] 

439 self._archive_task_history(removed) 

440 del self._tasks[task_id] 

441 await self._save() 

442 self._hass.bus.async_fire( 

443 EVENT_TASK_DELETED, events.task_event_data(removed) 

444 )

446 def managed_task_orphaned(self, task: dict[str, Any]) -> bool: 

447 """Whether a managed task's owning integration is no longer present. 

448 

449 A task is orphaned when its ``managed_by.config_entry_id`` names a config 

450 entry that is not currently loaded — i.e. the integration was uninstalled, 

451 disabled, or is failing to set up. Without a recorded ``config_entry_id`` we 

452 can't prove the owner is gone, so we treat it as present (not orphaned) and 

453 rely on the ``force`` escape hatch for cleanup. See ``models.deletion_blocked``. 

454 """ 

455 managed_by = task.get("managed_by") 

456 if not isinstance(managed_by, dict): 

457 return False 

458 entry_id = managed_by.get("config_entry_id") 

459 if not entry_id: 

460 return False 

461 entry = self._hass.config_entries.async_get_entry(entry_id) 

462 return entry is None or entry.state is not ConfigEntryState.LOADED

480 async def detach_tasks_from_device(self, device_id: str) -> list[str]: 

481 """Clear ``device_id`` on every task pointing at *device_id*. 

482 

483 Used when a virtual asset device is deleted: its tasks fall back to being 

484 standalone rather than dangling against a now-removed device. Returns the 

485 ids of the tasks that changed (so the caller can decide whether the 

486 per-task entity set needs rebuilding). 

487 """ 

488 changed: list[str] = [] 

489 for tid, task in self._tasks.items(): 

490 if task.get("device_id") == device_id: 

491 task["device_id"] = None

493 if changed: 

494 await self._save() 

495 for tid in changed: 

496 self._hass.bus.async_fire( 

497 EVENT_TASK_UPDATED, 

498 events.task_event_data( 

499 self._tasks[tid], extra={"changed_fields": ["device_id"]} 

500 ), 

501 ) 

502 return changed

705 async def delete_asset(self, asset_id: str) -> dict[str, Any] | None: 

706 """Remove an asset; returns the removed asset (for device cleanup) or None. 

707 

708 Drops the tasks **derived** from this asset's wear parts, but only **unlinks** 

709 a user-owned task that was *manually* linked to one of its consumables (clearing 

710 the dangling part source, keeping the task and its history). Also detaches any 

711 child asset that named it as a parent (so the child becomes standalone). 

712 """ 

713 asset = self._assets.pop(asset_id, None) 

714 if asset is None: 

715 return None 

716 dropped_ids = set() 

717 unlinked = [] 

718 for tid, t in self._tasks.items(): 

719 # An auto-created buy task for this appliance's part goes with it (it 

720 # carries a ``buy`` source, not ``part``, so the loop below would skip it). 

721 buy = _buy_source(t) 

722 if buy is not None and buy.get("asset_id") == asset_id: 

723 dropped_ids.add(tid) 

724 continue 

725 src = _part_source(t) 

726 if src is None or src.get("asset_id") != asset_id: 

727 continue 

728 if _is_manual_part_link(t): 

729 # A user-owned manual link: keep the task, just clear the now-dangling 

730 # link (its consumable is gone with the appliance). 

731 t["source"] = None 

732 unlinked.append(t) 

733 else: 

734 dropped_ids.add(tid) 

735 dropped = [self._tasks[tid] for tid in dropped_ids] 

736 self._tasks = { 

737 tid: t for tid, t in self._tasks.items() if tid not in dropped_ids 

738 } 

739 for child in self._assets.values(): 

740 if child.get("parent_asset_id") == asset_id: 

741 child["parent_asset_id"] = None 

742 await self._save() 

743 # Drop any uploaded-document blobs the appliance owned (best-effort cleanup). 

744 from . import manuals # lazy import: avoids a load-time import cycle 

745 

746 await manuals.async_delete_asset_documents(self._hass, asset_id) 

747 # The appliance and the wear-part tasks it owned are both gone — announce each. 

748 for task in dropped: 

749 self._hass.bus.async_fire(EVENT_TASK_DELETED, events.task_event_data(task)) 

750 # A surviving manual-link task lost its consumable — announce the source change. 

751 for task in unlinked: 

752 self._hass.bus.async_fire( 

753 EVENT_TASK_UPDATED, 

754 events.task_event_data(task, extra={"changed_fields": ["source"]}), 

755 ) 

756 self._hass.bus.async_fire(EVENT_ASSET_DELETED, events.asset_event_data(asset)) 

757 return asset

759 async def reconcile_part_tasks(self) -> bool: 

760 """Create/update/remove the maintenance tasks derived from wear parts. 

761 

762 Delegates the (pure) computation to :func:`reconcile.reconcile_part_tasks` 

763 and persists the result. Returns ``True`` if any task changed. 

764 

765 The generated task name is localized to Home Assistant's configured language 

766 (``hass.config.language`` — the household's primary language) here, at the HA 

767 boundary, so the pure reconciler stays language-agnostic. A language change 

768 relocalizes every generated name as ordinary name drift on the next reconcile 

769 (the ``__init__`` EVENT_CORE_CONFIG_UPDATE listener reloads the entry to force 

770 one immediately). 

771 """ 

772 old_tasks = self._tasks 

773 name_template, appliance_fallback = resolve_wear_task_naming( 

774 self._hass.config.language 

775 ) 

776 new_tasks, changed = _reconcile_part_tasks( 

777 self._assets, 

778 old_tasks, 

779 name_template=name_template, 

780 appliance_fallback=appliance_fallback, 

781 now=dt_util.now(), 

782 ) 

783 if changed: 

784 # A part-derived task dropped here means its wear part was removed while 

785 # the appliance remains; preserve its history on the appliance. (Deleting 

786 # the whole appliance drops its derived tasks via delete_asset, before any 

787 # reconcile, so this only archives part removals — not appliance deletes.) 

788 removed = [task for tid, task in old_tasks.items() if tid not in new_tasks] 

789 for task in removed: 

790 if _part_source(task): 

791 self._archive_task_history(task) 

792 self._tasks = new_tasks 

793 await self._save() 

794 # Wear-part tasks are created/removed here, bypassing add_task/delete_task, 

795 # so fire their lifecycle events directly (else this whole class of task is 

796 # silent to automations). Reconcile-time *edits* are intentionally not fired 

797 # as updates — they're internal churn (name/interval re-derivation), not a 

798 # user action. 

799 for task in removed: 

800 self._hass.bus.async_fire( 

801 EVENT_TASK_DELETED, events.task_event_data(task) 

802 ) 

803 for tid, task in new_tasks.items(): 

804 if tid not in old_tasks: 

805 self._hass.bus.async_fire( 

806 EVENT_TASK_CREATED, events.task_event_data(task) 

807 ) 

808 # A manual consumable link whose part was removed has its source cleared by 

809 # the reconcile (the task survives standalone); announce the source change. 

810 for tid, task in new_tasks.items(): 

811 old = old_tasks.get(tid) 

812 if old is not None and old.get("source") != task.get("source"): 

813 self._hass.bus.async_fire( 

814 EVENT_TASK_UPDATED, 

815 events.task_event_data( 

816 task, extra={"changed_fields": ["source"]} 

817 ), 

818 ) 

819 return changed

841 now=dt_util.now(),

844 return False

913 async def complete_task( 

914 self, 

915 task_id: str, 

916 completed_at: Any | None = None, 

917 *, 

918 origin: str | None = None, 

919 metadata: dict[str, Any] | None = None, 

920 ) -> dict[str, Any]: 

921 """Mark a task completed and advance its recurrence. 

922 

923 *metadata* is the optional per-completion context (``note``/``cost``/ 

924 ``photo``/``who``); it is cleaned here via 

925 ``models.normalize_completion_metadata`` and recorded on the new history 

926 entry. The cleaned mapping is also echoed in the completion event so 

927 listeners see the same data that was stored. 

928 

929 For a part-derived task, also stamp the part's ``last_replaced`` so the 

930 appliance record reflects the maintenance. 

931 

932 Fires the ``home_keeper_task_completed`` event so external integrations can 

933 mirror the completion. This is the single chokepoint every completion surface 

934 funnels through (the to-do list, the device mark-done button, and the 

935 ``complete_task`` service), so firing here — rather than in the service handler 

936 — is what makes completion observable from anywhere. ``origin`` is an opaque, 

937 caller-supplied marker echoed back in the event purely so a contributing 

938 integration can ignore the echo of a completion it initiated; Home Keeper does 

939 not interpret it. 

940 """ 

941 existing = self._tasks.get(task_id) 

942 if existing is None: 

943 raise KeyError(task_id) 

944 _reject_synced_problem(existing, origin) 

945 now = dt_util.now() 

946 when = completed_at or now 

947 if when.tzinfo is None: 

948 # ``cv.datetime`` accepts offset-less strings as naive datetimes; 

949 # qualify with HA's configured zone (mirrors ``models._coerce_seed``) 

950 # so the event payload / part stamp / recurrence math all stay aware. 

951 when = when.replace(tzinfo=now.tzinfo) 

952 clean_metadata = models.normalize_completion_metadata(metadata) 

953 updated = recurrence.apply_completion( 

954 dict(existing), when, now=now, metadata=clean_metadata 

955 ) 

956 self._reset_usage_baseline(updated) 

957 self._tasks[task_id] = updated 

958 # A task carries at most one reserved source, so exactly one stock side-effect 

959 # applies: a part-linked completion *consumes* a spare, a buy reminder 

960 # *restocks*. Branch (rather than call both self-guarding helpers) so a 

961 # malformed both-sources task can never both consume and restock on one 

962 # completion — a `part` source wins, mirroring add_task's reserved-key guard. 

963 if _part_source(updated): 

964 self._stamp_part_replacement(updated, when) 

965 elif _buy_source(updated): 

966 self._stamp_buy_restock(updated) 

967 await self._save() 

968 _LOGGER.debug( 

969 "Completed task %s; next due %s", task_id, updated.get("next_due") 

970 ) 

971 self._hass.bus.async_fire( 

972 EVENT_TASK_COMPLETED, 

973 events.completion_event_data( 

974 updated, when, origin, metadata=clean_metadata 

975 ), 

976 ) 

977 return updated

1058 def _reset_usage_baseline(self, task: dict[str, Any]) -> None: 

1059 """Reset a usage sensor task's meter baseline to the live reading on completion. 

1060 

1061 Completing a usage task ("I changed the oil") resets its counter just as 

1062 completing a floating task resets its clock: the next arming is measured from 

1063 the reading at completion. Done here (the store is HA-aware) rather than in the 

1064 pure recurrence layer. If the bound entity is currently unavailable the baseline 

1065 is cleared (``None``) so the watcher re-anchors on the first valid reading after 

1066 completion rather than measuring from the stale pre-completion baseline. 

1067 """ 

1068 cfg = sensor_tasks.sensor_config(task) 

1069 if cfg is None or cfg.get("mode") != SENSOR_MODE_USAGE: 

1070 return 

1071 # Reset the meter to the reading at completion. If the entity is currently 

1072 # unavailable, clear the baseline (``None``) so the watcher re-anchors to the 

1073 # first valid reading *after* completion rather than measuring from the now 

1074 # stale pre-completion baseline — which could otherwise immediately re-arm the 

1075 # task if the meter advanced past target while the entity was unavailable. 

1076 cfg["baseline"] = sensor_watcher.read_sensor_value(self._hass, cfg)

1094 def _stamp_buy_restock(self, task: dict[str, Any]) -> None: 

1095 """On completing an auto-created buy task, restock its part. 

1096 

1097 Adds the part's ``restock_quantity`` (default 1) to ``stock`` and emits the 

1098 resulting stock transition — normally a ``restocked`` crossing, which the next 

1099 buy-task reconcile turns into removal of this now-satisfied reminder. Buy tasks 

1100 never carry a ``part`` source, so ``_stamp_part_replacement`` leaves them 

1101 untouched (no double-mutation). 

1102 """ 

1103 src = _buy_source(task) 

1104 if not src: 

1105 return 

1106 asset = self._assets.get(src["asset_id"]) 

1107 if not asset: 

1108 return 

1109 for part in asset.get("parts", []): 

1110 if part.get("id") == src.get("part_id"): 

1111 qty = max(1, int(part.get("restock_quantity") or 1)) 

1112 self._emit_stock_event(assets.adjust_part_stock(part, qty), asset, part) 

1113 break 

1114 

1115 def _emit_stock_event( 

1116 self, transition: str, asset: dict[str, Any], part: dict[str, Any] 

1117 ) -> None: 

1118 """Fire the bus event for a stock *transition* (low / out / restocked), if any. 

1119 

1120 Edge-triggered: ``assets.stock_transition`` returns the crossing this change 

1121 caused (or ``none``), so users can automate a reorder / shopping-list add / 

1122 "back in stock" notification without Home Keeper owning a shopping integration. 

1123 """ 

1124 event = _STOCK_EVENT.get(transition) 

1125 if event is not None: 

1126 self._hass.bus.async_fire(event, events.stock_event_data(asset, part)) 

1127 

1128 async def adjust_part_stock( 

1129 self, asset_id: str, part_id: str, delta: int 

1130 ) -> dict[str, Any]: 

1131 """Change a part's on-hand spare count by ``delta`` (clamped at zero). 

1132 

1133 Persists and fires the matching edge-triggered stock event once when the 

1134 adjustment crosses a threshold — low-stock, out-of-stock, or restocked (a 

1135 decrease while already low never nags). Returns the updated asset. Raises 

1136 ``KeyError`` for an unknown asset or part. 

1137 """ 

1138 asset = self._assets.get(asset_id) 

1139 if asset is None: 

1140 raise KeyError(asset_id) 

1141 for part in asset.get("parts", []): 

1142 if part.get("id") == part_id: 

1143 transition = assets.adjust_part_stock(part, delta) 

1144 await self._save() 

1145 self._emit_stock_event(transition, asset, part) 

1146 return asset 

1147 raise KeyError(part_id)
