55def _reject_synced_problem(task: dict[str, Any], origin: str | None) -> None: 

56 """Raise unless *origin* authorizes mutating a problem-sensor-synced task. 

57 

58 A synced task mirrors a ``device_class: problem`` binary sensor and is owned by 

59 the sync. Every user-facing surface (to-do, button, service, websocket, panel) 

60 calls without :data:`ORIGIN_PROBLEM_SENSOR_SYNC`, so this rejects them with a 

61 clear message; only the internal sync (which passes the marker) may arm or clear 

62 the task. The problem itself has to be resolved in the originating integration. 

63 """ 

64 if _problem_source(task) is None or origin == ORIGIN_PROBLEM_SENSOR_SYNC: 

65 return 

66 entity_id = _problem_entity(task) or "the originating integration" 

67 raise models.TaskValidationError( 

68 f"This task mirrors the problem sensor {entity_id} and can't be cleared in " 

69 "Home Keeper. Resolve the problem in the originating integration — the task " 

70 "clears automatically when the sensor returns to OK." 

71 )

74def _changed_fields(before: dict[str, Any], after: dict[str, Any]) -> list[str]: 

75 """Top-level keys whose value differs between *before* and *after*. 

76 

77 Drives the ``changed_fields`` payload on the update events (and the decision to 

78 suppress a no-op update event entirely). Backend-managed bookkeeping keys that 

79 aren't user-meaningful are ignored so a reschedule doesn't spam ``next_due``/ 

80 ``completions`` churn as "changes". 

81 """ 

82 ignore = {"completions", "last_completed", "next_due", "created"} 

83 keys = (set(before) | set(after)) - ignore 

84 return sorted(k for k in keys if before.get(k) != after.get(k))

97 """Load tasks and assets from disk (no-op safe on first run). 

98

107 if data and isinstance(data.get("assets"), dict): 

108 self._assets = data["assets"]

114 changed = False 

115 for asset in self._assets.values(): 

116 if assets.migrate_legacy_part_numbers(asset): 

117 changed = True 

118 if self._clean_relationship_links(): 

119 changed = True 

120 if changed: 

121 await self._save()

181 async def trigger_task( 

182 self, task_id: str, *, origin: str | None = None 

183 ) -> dict[str, Any]: 

184 """Arm a condition-driven (``triggered``) task so it reads as due-now. 

185 

186 Sets ``next_due`` to now, moving the task from dormant to active. This is the 

187 owner-facing counterpart to ``complete_task`` (which clears it back to 

188 dormant): an integration calls it when the condition it monitors becomes true 

189 again (e.g. a battery drops low after a prior replacement). Idempotent — 

190 arming an already-active task just refreshes its trigger time. Only valid for 

191 ``triggered`` tasks; rejects others so callers don't accidentally strand a 

192 scheduled task on a fixed due date. 

193 """ 

194 existing = self._tasks.get(task_id) 

195 if existing is None: 

196 raise KeyError(task_id) 

197 if existing.get("recurrence_type") != REC_TRIGGERED: 

198 raise models.TaskValidationError( 

199 "trigger_task is only valid for triggered (condition-driven) tasks" 

200 ) 

201 _reject_synced_problem(existing, origin) 

202 existing["next_due"] = dt_util.now().isoformat() 

203 await self._save() 

204 _LOGGER.debug("Triggered task %s (armed, due now)", task_id) 

205 self._hass.bus.async_fire( 

206 EVENT_TASK_TRIGGERED, events.task_event_data(existing) 

207 ) 

208 return existing

210 async def delete_task(self, task_id: str, *, force: bool = False) -> None: 

211 task = self._tasks.get(task_id) 

212 if task is not None and _part_source(task): 

213 # Derived from a wear part; deleting it here would just be recreated by 

214 # the next reconcile. Direct the user to manage the part instead. 

215 raise models.TaskValidationError( 

216 "This task is managed by an appliance wear part; remove or change " 

217 "the part to delete it." 

218 ) 

219 if task is not None: 

220 managed_by = task.get("managed_by") 

221 orphaned = self.managed_task_orphaned(task) 

222 if models.deletion_blocked(task, orphaned=orphaned, force=force): 

223 display_name = (managed_by or {}).get( 

224 "display_name" 

225 ) or "an integration" 

226 raise models.TaskValidationError( 

227 f"This task is managed by {display_name}. " 

228 f"Delete it from {display_name} instead." 

229 ) 

230 if task_id in self._tasks: 

231 removed = self._tasks[task_id] 

232 self._archive_task_history(removed) 

233 del self._tasks[task_id] 

234 await self._save() 

235 self._hass.bus.async_fire( 

236 EVENT_TASK_DELETED, events.task_event_data(removed) 

237 )

239 def managed_task_orphaned(self, task: dict[str, Any]) -> bool: 

240 """Whether a managed task's owning integration is no longer present. 

241 

242 A task is orphaned when its ``managed_by.config_entry_id`` names a config 

243 entry that is not currently loaded — i.e. the integration was uninstalled, 

244 disabled, or is failing to set up. Without a recorded ``config_entry_id`` we 

245 can't prove the owner is gone, so we treat it as present (not orphaned) and 

246 rely on the ``force`` escape hatch for cleanup. See ``models.deletion_blocked``. 

247 """ 

248 managed_by = task.get("managed_by") 

249 if not isinstance(managed_by, dict): 

250 return False 

251 entry_id = managed_by.get("config_entry_id") 

252 if not entry_id: 

253 return False 

254 entry = self._hass.config_entries.async_get_entry(entry_id) 

255 return entry is None or entry.state is not ConfigEntryState.LOADED

257 def _archive_task_history(self, task: dict[str, Any]) -> None: 

258 """Preserve a deleted task's completion history on its appliance, if any. 

259 

260 Reference-counting retention: a task's history outlives the task only while 

261 an appliance still references it. A standalone task's history is dropped 

262 with it (``find_archiving_asset`` returns None). Mutates the asset in place; 

263 the caller persists via ``_save``. 

264 """ 

265 asset = assets.find_archiving_asset(self._assets, task) 

266 if asset is None: 

267 return 

268 entry = assets.build_archived_history( 

269 task, archived_at=dt_util.now().isoformat() 

270 ) 

271 assets.append_task_history(asset, entry)

273 async def detach_tasks_from_device(self, device_id: str) -> list[str]: 

274 """Clear ``device_id`` on every task pointing at *device_id*. 

275 

276 Used when a virtual asset device is deleted: its tasks fall back to being 

277 standalone rather than dangling against a now-removed device. Returns the 

278 ids of the tasks that changed (so the caller can decide whether the

293 ), 

294 ) 

295 return changed

374 async def delete_asset(self, asset_id: str) -> dict[str, Any] | None: 

375 """Remove an asset; returns the removed asset (for device cleanup) or None. 

376 

377 Also drops tasks derived from this asset's wear parts and detaches any 

378 child asset that named it as a parent (so the child becomes standalone). 

379 """ 

380 asset = self._assets.pop(asset_id, None) 

381 if asset is None: 

382 return None 

383 dropped_ids = set() 

384 for tid, t in self._tasks.items(): 

385 src = _part_source(t) 

386 if src is not None and src.get("asset_id") == asset_id: 

387 dropped_ids.add(tid) 

388 dropped = [self._tasks[tid] for tid in dropped_ids] 

389 self._tasks = { 

390 tid: t for tid, t in self._tasks.items() if tid not in dropped_ids 

391 } 

392 for child in self._assets.values(): 

393 if child.get("parent_asset_id") == asset_id: 

394 child["parent_asset_id"] = None 

395 await self._save() 

396 # The appliance and the wear-part tasks it owned are both gone — announce each. 

397 for task in dropped: 

398 self._hass.bus.async_fire(EVENT_TASK_DELETED, events.task_event_data(task)) 

399 self._hass.bus.async_fire(EVENT_ASSET_DELETED, events.asset_event_data(asset)) 

400 return asset

402 async def reconcile_part_tasks(self) -> bool: 

403 """Create/update/remove the maintenance tasks derived from wear parts. 

404 

405 Delegates the (pure) computation to :func:`reconcile.reconcile_part_tasks` 

406 and persists the result. Returns ``True`` if any task changed. 

407 """ 

408 old_tasks = self._tasks 

409 new_tasks, changed = _reconcile_part_tasks( 

410 self._assets, old_tasks, now=dt_util.now() 

411 ) 

412 if changed: 

413 # A part-derived task dropped here means its wear part was removed while 

414 # the appliance remains; preserve its history on the appliance. (Deleting 

415 # the whole appliance drops its derived tasks via delete_asset, before any 

416 # reconcile, so this only archives part removals — not appliance deletes.) 

417 removed = [task for tid, task in old_tasks.items() if tid not in new_tasks] 

418 for task in removed: 

419 if _part_source(task): 

420 self._archive_task_history(task) 

421 self._tasks = new_tasks 

422 await self._save() 

423 # Wear-part tasks are created/removed here, bypassing add_task/delete_task, 

424 # so fire their lifecycle events directly (else this whole class of task is 

425 # silent to automations). Reconcile-time *edits* are intentionally not fired 

426 # as updates — they're internal churn (name/interval re-derivation), not a 

427 # user action. 

428 for task in removed: 

429 self._hass.bus.async_fire( 

430 EVENT_TASK_DELETED, events.task_event_data(task) 

431 ) 

432 for tid, task in new_tasks.items(): 

433 if tid not in old_tasks: 

434 self._hass.bus.async_fire( 

435 EVENT_TASK_CREATED, events.task_event_data(task) 

436 ) 

437 return changed

439 async def reconcile_problem_sensor_tasks( 

440 self, eligible: dict[str, dict[str, Any]], *, config_entry_id: str 

441 ) -> bool: 

442 """Sync the triggered tasks mirroring ``device_class: problem`` sensors. 

443 

444 *eligible* maps ``entity_id`` -> ``{"name", "device_id", "area_id", 

445 "is_problem"}`` for the sensors that should be synced (the HA-aware filtering 

446 — option on, exclusions, skipping Home Keeper's own entities — happens in 

447 ``problem_sync.py``; an empty map removes every synced task, e.g. when the 

448 option is turned off). Delegates the diff to the pure 

449 :func:`problem_tasks.reconcile_problem_tasks`, persists, and fires the 

450 matching lifecycle events. Returns ``True`` when the per-task **entity set** 

451 changed (a task was created or removed) so the caller can decide between a 

452 full entry reload and a plain coordinator refresh. 

453 """ 

454 new_tasks, ops, changed = _reconcile_problem_tasks( 

455 eligible, self._tasks, config_entry_id=config_entry_id, now=dt_util.now() 

456 ) 

457 if not changed: 

458 return False 

459 self._tasks = new_tasks 

460 await self._save() 

461 entity_set_changed = False 

462 for kind, task in ops: 

463 if kind == "created": 

464 self._hass.bus.async_fire( 

465 EVENT_TASK_CREATED, events.task_event_data(task) 

466 ) 

467 entity_set_changed = True 

468 elif kind == "deleted": 

469 self._hass.bus.async_fire( 

470 EVENT_TASK_DELETED, events.task_event_data(task) 

471 ) 

472 entity_set_changed = True 

473 elif kind == "armed": 

474 self._hass.bus.async_fire( 

475 EVENT_TASK_TRIGGERED, events.task_event_data(task) 

476 ) 

477 elif kind == "cleared": 

478 self._hass.bus.async_fire( 

479 EVENT_TASK_COMPLETED, 

480 events.completion_event_data( 

481 task, dt_util.now(), ORIGIN_PROBLEM_SENSOR_SYNC 

482 ), 

483 ) 

484 return entity_set_changed

486 async def complete_task( 

487 self, 

488 task_id: str, 

489 completed_at: Any | None = None, 

490 *, 

491 origin: str | None = None, 

492 ) -> dict[str, Any]: 

493 """Mark a task completed and advance its recurrence. 

494 

495 For a part-derived task, also stamp the part's ``last_replaced`` so the 

496 appliance record reflects the maintenance. 

497 

498 Fires the ``home_keeper_task_completed`` event so external integrations can 

499 mirror the completion. This is the single chokepoint every completion surface 

500 funnels through (the to-do list, the device mark-done button, and the 

501 ``complete_task`` service), so firing here — rather than in the service handler 

502 — is what makes completion observable from anywhere. ``origin`` is an opaque, 

503 caller-supplied marker echoed back in the event purely so a contributing 

504 integration can ignore the echo of a completion it initiated; Home Keeper does 

505 not interpret it. 

506 """ 

507 existing = self._tasks.get(task_id) 

508 if existing is None: 

509 raise KeyError(task_id) 

510 _reject_synced_problem(existing, origin) 

511 now = dt_util.now() 

512 when = completed_at or now 

513 updated = recurrence.apply_completion(dict(existing), when, now=now) 

514 self._tasks[task_id] = updated 

515 self._stamp_part_replacement(updated, when) 

516 await self._save() 

517 _LOGGER.debug( 

518 "Completed task %s; next due %s", task_id, updated.get("next_due") 

519 ) 

520 self._hass.bus.async_fire( 

521 EVENT_TASK_COMPLETED, 

522 events.completion_event_data(updated, when, origin), 

523 ) 

524 return updated

560 def _stamp_part_replacement(self, task: dict[str, Any], when: Any) -> None: 

561 src = _part_source(task) 

562 if not src: 

563 return 

564 asset = self._assets.get(src["asset_id"]) 

565 if not asset: 

566 return 

567 when_date = when.date().isoformat() if hasattr(when, "date") else str(when)[:10] 

568 for part in asset.get("parts", []): 

569 if part.get("id") == src.get("part_id"): 

570 part["last_replaced"] = when_date 

571 # Completing a wear-part replacement consumes one stocked spare; 

572 # signal a low/out-of-stock crossing so users can automate a reorder. 

573 self._emit_stock_event(assets.consume_part_stock(part), asset, part) 

574 break 

575 

576 def _emit_stock_event( 

577 self, transition: str, asset: dict[str, Any], part: dict[str, Any] 

578 ) -> None: 

579 """Fire the bus event for a stock *transition* (low / out / restocked), if any. 

580 

581 Edge-triggered: ``assets.stock_transition`` returns the crossing this change 

582 caused (or ``none``), so users can automate a reorder / shopping-list add / 

583 "back in stock" notification without Home Keeper owning a shopping integration. 

584 """ 

585 event = _STOCK_EVENT.get(transition) 

586 if event is not None: 

587 self._hass.bus.async_fire(event, events.stock_event_data(asset, part)) 

588 

589 async def adjust_part_stock( 

590 self, asset_id: str, part_id: str, delta: int 

591 ) -> dict[str, Any]: 

592 """Change a part's on-hand spare count by ``delta`` (clamped at zero). 

593 

594 Persists and fires the matching edge-triggered stock event once when the 

595 adjustment crosses a threshold — low-stock, out-of-stock, or restocked (a 

596 decrease while already low never nags). Returns the updated asset. Raises 

597 ``KeyError`` for an unknown asset or part. 

598 """ 

599 asset = self._assets.get(asset_id) 

600 if asset is None: 

601 raise KeyError(asset_id) 

602 for part in asset.get("parts", []): 

603 if part.get("id") == part_id: 

604 transition = assets.adjust_part_stock(part, delta) 

605 await self._save() 

606 self._emit_stock_event(transition, asset, part) 

607 return asset 

608 raise KeyError(part_id)
