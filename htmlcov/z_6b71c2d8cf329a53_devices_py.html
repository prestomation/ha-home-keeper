1"""Registry-device provisioning for Home Keeper assets. 

2 

3This is the *virtual-device* half of the asset feature (see ``IDEAS.md`` / 

4``docs/DESIGN.md``): when an appliance has no Home Assistant device to attach 

5maintenance tasks to, Home Keeper registers a real device-registry entry for it so 

6tasks, future batteries, and asset metadata all converge on one device page. 

7 

8Devices we create are tied to our config entry, so Home Assistant removes them 

9automatically when the integration is removed — no residue. Reconciliation is 

10idempotent: it can run on every setup and after every asset mutation. 

11"""

51def _ancestor_depth(store: HomeKeeperStore, asset: dict[str, Any]) -> int: 

52 """Number of parent links above *asset* (used to provision parents first).""" 

53 depth = 0 

54 seen: set[str] = {asset["id"]} 

55 cursor = asset.get("parent_asset_id") 

56 while cursor and cursor not in seen: 

57 seen.add(cursor) 

58 depth += 1 

59 parent = store.get_asset(cursor) 

60 cursor = parent.get("parent_asset_id") if parent else None 

61 if cursor: # would_create_cycle should prevent this; surface corrupt storage. 

62 _LOGGER.error("Cyclic parent chain detected at asset %s", asset["id"]) 

63 return depth

66def _is_asset_device(device: dr.DeviceEntry) -> bool: 

67 """True if *device* is one of our provisioned virtual asset devices. 

68 

69 Virtual asset devices carry an ``asset_``-prefixed identifier; per-task 

70 self-owned devices key on the bare (uuid) task id, so they never match. 

71 """ 

72 return any( 

73 domain == DOMAIN and ident.startswith(f"{ASSET_IDENTIFIER_PREFIX}_") 

74 for domain, ident in device.identifiers 

75 )

78def _supports_kwarg(func: Any, name: str) -> bool: 

79 try: 

80 return name in inspect.signature(func).parameters 

81 except (TypeError, ValueError): # pragma: no cover - builtins without signatures 

82 # Be conservative: if we can't confirm support, don't pass the kwarg. 

83 return False

86async def async_apply_asset_change( 

87 hass: HomeAssistant, entry: ConfigEntry, store: HomeKeeperStore 

88) -> None: 

89 """Reconcile devices then reload the entry after an asset mutation. 

90 

91 Shared by the service and websocket handlers so both refresh the registry and 

92 the per-asset/per-task entity set identically. The reload re-runs setup 

93 reconciliation, but that pass is idempotent (get_or_create returns the existing 

94 device; snapshot writes are skipped when unchanged). 

95 """ 

96 await async_reconcile_assets(hass, entry, store) 

97 # Wear parts may have created/removed derived maintenance tasks; sync them 

98 # before the reload rebuilds the per-task entity set. 

99 await store.reconcile_part_tasks() 

100 await hass.config_entries.async_reload(entry.entry_id)

103async def async_reconcile_assets( 

104 hass: HomeAssistant, entry: ConfigEntry, store: HomeKeeperStore 

105) -> None: 

106 """Make the device registry match the current set of assets. 

107 

108 * Virtual assets get a registry device created (idempotently) and kept in sync 

109 with their editable fields; the assigned ``device.id`` is written back to the 

110 asset so tasks/metadata entities can resolve it. 

111 * Existing-device assets get their identifiers/connections snapshotted for 

112 reconciliation, and a warning if the referenced device has gone away. 

113 * Orphan asset devices (ours, but no longer backed by an asset) are removed. 

114 """ 

115 registry = dr.async_get(hass) 

116 wanted_identifiers: set[tuple[str, str]] = set() 

117 dirty = False 

118 

119 # Provision parents before children so a subdevice's via_device parent already 

120 # has a resolved device id when we link it. 

121 for asset in sorted(store.list_assets(), key=lambda a: _ancestor_depth(store, a)): 

122 if asset.get("kind") == ASSET_KIND_VIRTUAL: 

123 # _reconcile_virtual persists its own device_id write-back. 

124 await _reconcile_virtual(hass, entry, store, registry, asset) 

125 wanted_identifiers.add(asset_model.asset_device_identifier(asset["id"])) 

126 elif _reconcile_existing(registry, asset): 

127 dirty = True 

128 

129 # In-place edits to existing-device assets (recovered device_id, refreshed 

130 # identifiers/connections snapshot) must be flushed to disk or they're lost on 

131 # restart and snapshot recovery can never work. 

132 if dirty: 

133 await store.async_persist() 

134 

135 # Prune asset devices we own that no longer correspond to an asset. Guard 

136 # against ever removing a per-task self-owned device — those key on the bare 

137 # task id — by excluding any device that matches a current task's identifier, 

138 # even in the (uuid-impossible) case a task id collides with the asset prefix. 

139 task_identifiers = {(DOMAIN, tid) for tid in store.get_tasks()} 

140 for device in dr.async_entries_for_config_entry(registry, entry.entry_id): 

141 if ( 

142 _is_asset_device(device) 

143 and not (device.identifiers & wanted_identifiers) 

144 and not (device.identifiers & task_identifiers) 

145 ): 

146 _LOGGER.debug("Removing orphaned asset device %s", device.id) 

147 registry.async_remove_device(device.id)

150async def _reconcile_virtual( 

151 hass: HomeAssistant, 

152 entry: ConfigEntry, 

153 store: HomeKeeperStore, 

154 registry: dr.DeviceRegistry, 

155 asset: dict[str, Any], 

156) -> None: 

157 identifier = asset_model.asset_device_identifier(asset["id"]) 

158 

159 # Resolve the native via_device parent (only our own virtual subdevices). 

160 parent_asset_id = asset.get("parent_asset_id") or None 

161 via_device = ( 

162 asset_model.asset_device_identifier(parent_asset_id) 

163 if parent_asset_id 

164 else None 

165 ) 

166 parent = store.get_asset(parent_asset_id) if parent_asset_id else None 

167 parent_device_id = parent.get("device_id") if parent else None 

168 

169 create_kwargs: dict[str, Any] = { 

170 "config_entry_id": entry.entry_id, 

171 "identifiers": {identifier}, 

172 "name": asset["name"], 

173 "manufacturer": asset.get("manufacturer") or None, 

174 "model": asset.get("model") or None, 

175 "configuration_url": _CONFIGURATION_URL, 

176 } 

177 if via_device is not None: 

178 create_kwargs["via_device"] = via_device 

179 device = registry.async_get_or_create(**create_kwargs) 

180 

181 # Keep the registry in sync with subsequent edits. 

182 updates: dict[str, Any] = {} 

183 if device.name != asset["name"]: 

184 updates["name"] = asset["name"] 

185 for field in ("manufacturer", "model"): 

186 desired = asset.get(field) or None 

187 if getattr(device, field, None) != desired and _supports_kwarg( 

188 registry.async_update_device, field 

189 ): 

190 updates[field] = desired 

191 # Note the absence of a truthy guard on area_id: passing None propagates an 

192 # area *clear* to the device (a truthy-only check would strand the old area). 

193 # Guard against a dangling area_id (the area was deleted in HA after it was 

194 # assigned): async_update_device rejects an unknown area, so self-heal to None. 

195 area_id = asset.get("area_id") or None 

196 if area_id and not area_exists(hass, area_id): 

197 area_id = None 

198 if area_id != device.area_id: 

199 updates["area_id"] = area_id 

200 if device.configuration_url != _CONFIGURATION_URL: 

201 updates["configuration_url"] = _CONFIGURATION_URL 

202 # Re-parent / un-parent after creation (via_device on create only applies the 

203 # first time). via_device_id is the parent's *device id*, which parents-first 

204 # ordering has already resolved. 

205 if _supports_kwarg(registry.async_update_device, "via_device_id"): 

206 if device.via_device_id != parent_device_id: 

207 updates["via_device_id"] = parent_device_id 

208 elif parent_asset_id and device.via_device_id is None: 

209 _LOGGER.warning( 

210 "This Home Assistant version can't update a device's parent after " 

211 "creation; subdevice %s may not nest under its parent", 

212 asset["id"], 

213 ) 

214 if updates: 

215 updated = registry.async_update_device(device.id, **updates) 

216 if updated is not None: 

217 device = updated 

218 

219 await store.set_asset_device_id(asset["id"], device.id)

222def _reconcile_existing(registry: dr.DeviceRegistry, asset: dict[str, Any]) -> bool: 

223 """Resolve an existing-device asset and refresh its snapshot in place. 

224 

225 Returns ``True`` if the asset dict was mutated (so the caller persists it). 

226 """ 

227 device_id = asset.get("device_id") 

228 device = registry.async_get(device_id) if device_id else None 

229 changed = False 

230 if device is None: 

231 # The referenced device may have been recreated under a new id by its 

232 # owning integration; try to recover it from the stored snapshot. 

233 device = _resolve_by_snapshot(registry, asset) 

234 if device is None: 

235 _LOGGER.warning( 

236 "Home Keeper asset %s references missing device %s; metadata " 

237 "entities will not appear until the device returns", 

238 asset["id"], 

239 device_id, 

240 ) 

241 return False 

242 asset["device_id"] = device.id 

243 changed = True 

244 # Refresh the reconciliation snapshot from the live device (only marking the 

245 # asset dirty when it actually changed, to avoid needless writes). 

246 identifiers = [list(i) for i in device.identifiers] 

247 connections = [list(c) for c in device.connections] 

248 if asset.get("identifiers") != identifiers: 

249 asset["identifiers"] = identifiers 

250 changed = True 

251 if asset.get("connections") != connections: 

252 asset["connections"] = connections 

253 changed = True 

254 return changed

257def _resolve_by_snapshot( 

258 registry: dr.DeviceRegistry, asset: dict[str, Any] 

259) -> dr.DeviceEntry | None: 

260 """Find a device matching the asset's stored identifiers/connections.""" 

261 for ident in asset.get("identifiers", []): 

262 device = registry.async_get_device(identifiers={tuple(ident)}) 

263 if device is not None: 

264 return device 

265 connections = {tuple(c) for c in asset.get("connections", [])} 

266 if connections: 

267 return registry.async_get_device(connections=connections) 

268 return None

271async def async_remove_asset_device( 

272 hass: HomeAssistant, asset: dict[str, Any] 

273) -> str | None: 

274 """Remove the virtual device backing *asset*; returns its device id if removed. 

275 

276 No-op for existing-device assets (we never owned that device). Tasks attached 

277 to a removed virtual device should be detached by the caller. 

278 """ 

279 if asset.get("kind") != ASSET_KIND_VIRTUAL: 

280 return None 

281 registry = dr.async_get(hass) 

282 identifier = asset_model.asset_device_identifier(asset["id"]) 

283 device = registry.async_get_device(identifiers={identifier}) 

284 if device is not None: 

285 registry.async_remove_device(device.id) 

286 return device.id 

287 return asset.get("device_id")
